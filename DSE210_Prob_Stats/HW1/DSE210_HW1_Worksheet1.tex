\begin{homeworkProblem}

\begin{enumerate}
\item A = \{a, b, c, d, e\}
\item $A^3 = A \times A \times A$
\item $5^3 = 5 \times 5 \times 5 = \boxed{125}$
\end{enumerate}

\end{homeworkProblem}

\begin{homeworkProblem}
There are $\boxed{2^{500}}$ binary sequences of length 500.
\end{homeworkProblem}

\begin{homeworkProblem}
\textit{Given:} \textit{A} and \textit{B} are sets with $\abs{A} = 3$ and $\abs{B} = 4$.

\begin{enumerate}
\item $\abs{A \cup B} = \boxed{7}$ if $A \cap B = \varnothing$
\item $\abs{A \cup B} = \boxed{4}$ if $A \subset B$
\item $\abs{A \cap B} = \boxed{3}$ if $A \subset B$, and $\abs{A \cap B} = \boxed{0}$ if $A \cap B = \varnothing$

\end{enumerate}

\end{homeworkProblem}

\begin{homeworkProblem}
There are $4! = 4 \times 3 \times 2 \times 1 = \boxed{24}$ different orderings.
\end{homeworkProblem}

\begin{homeworkProblem}
$\Perm{26}{5} = \cfrac{26!}{(26-5)!} = \cfrac{26!}{21!} = 26 \times 25 \times 24 \times 23 \times 22 = \boxed{7,893,600}$
\end{homeworkProblem}

\begin{homeworkProblem}
$\Comb{10}{3} = \cfrac{10!}{(10-3)! 3!} = \cfrac{10!}{7! 3!} = \cfrac{10 \times 9 \times 8}{3 \times 2 \times 1} = \cfrac{720}{6} = \boxed{120}$
\end{homeworkProblem}

\begin{homeworkProblem}
$\Perm{10}{5} = \cfrac{10!}{(10-5)!} = \cfrac{10!}{5!} = 10 \times 9 \times 8 \times 7 \times 6 = \boxed{30,240}$
\end{homeworkProblem}
